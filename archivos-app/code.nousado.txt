const cantidadData = [
  { producto: "Lupa", total_cantidad: 10 },
  { producto: "Helado", total_cantidad: 25 },
  { producto: "Martillo", total_cantidad: 15 }
];

const ingresosData = [
  { producto: "Lupa", ingresos: 490.00 },
  { producto: "Helado", ingresos: 500.00 },
  { producto: "Martillo", ingresos: 750.00 }
];

// Gráfico de cantidad vendida


new Chart(document.getElementById('cantidadChart'), {
  type: 'bar',
  data: {
    labels: cantidadData.map(d => d.producto),
    datasets: [{
      label: 'Cantidad vendida',
      data: cantidadData.map(d => d.total_cantidad),
      backgroundColor: 'rgba(54, 162, 235, 0.6)'
    }]
  }
});

// Gráfico de ingresos por producto
new Chart(document.getElementById('ingresosChart'), {
  type: 'pie',
  data: {
    labels: ingresosData.map(d => d.producto),
    datasets: [{
      label: 'Ingresos',
      data: ingresosData.map(d => d.ingresos),
      backgroundColor: ['#FF6384', '#36A2EB', '#FFCE56']
    }]
  }
});
